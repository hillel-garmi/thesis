\section{Proof that the Galois group of any irreducible factor of $x^n +ax + b$ is symmetric for different small values of $a$ and $b$} 
\localtableofcontents
\subsection{Summary of results}
Fix $n > 0$ and $a,b \neq 0$ and let 
\[
  f = x^n + ax + b
\]
If all the conditions in one of the rows of the table below are satisfied, then the Galois group of any irreducible factor of $f$ is the full symmetric group.

\begin{center}
\begin{tabular}{||c c c c||} 
 \hline
 $a = $ & $b = $ & $n \not \equiv$ & $\mod$ \\ [0.5ex] 
 \hline\hline
 $\pm 1$ & $ \pm 1$ & & \\ 
 \hline
 $\pm 2$ & $1$ & & \\ 
 \hline
 $\pm 2$ & $\pm 2$ & & \\ 
 \hline
 $\pm 3$ & $-3$ & & \\ 
 \hline
 $\pm 4$ & $-2$ & & \\ 
 \hline
 $\pm 3$ & $1$ & $-a$ & 18 \\
 \hline
 $\pm 4$  & $1$ & $4$ & $8$ \\
 \hline
 $5$  & $1$ & $10,40,45,60,90,95$ & $100$\\
 \hline
 $-5$  & $1$ & $5,10,40,55,60,90$ & $100$ \\ [1ex] 
 \hline
\end{tabular}
\end{center}

\subsection{Notations}
Fix an algebraic closure $\overline{\Q}$ of $\Q$ and for each prime $p$, fix an algebraic closure $\overline{\Q_p}$ of $\Q_p$ and an embedding $\overline{\Q} \hookrightarrow \overline{\Q_p}$ extending the natural embedding $\Q \hookrightarrow \Q_p$.
Denote by $G$ the absolute Galois group $\Gal\left( \overline{\Q}/\Q \right)$, and for each prime $p$, denote by $I\left( p \right) \subseteq G$ the absolute inertia group at $p$, i.e., the embedding into $G$ of the inertia group $\Gal\left( \overline{\Q_p}/\Q_p^{\un} \right)$, where $\Q_p^{\un}$ is the maximal unramified extension of $\Q_p$ inside $\overline{\Q_p}$.

Fix an irreducible factor $g$ of $f$ (which most of the time will be $f$ itself) and a root $\alpha \in \overline{\Q}$ of $g$ and denote $L = \Q\left( \alpha \right)$.

For each polynomial $h$ and number field $K$, we will denote by $\d\left( h \right)$ and $\d\left( K \right)$ \editorial{(or for legacy reasons also $\disc\left( K \right)$)} the discriminant of $h$ and of $K$ respectively and by $\rd\left( K \right)$ the root discriminant of $K$, i.e. $\rd\left( K \right) = \d\left( K \right)^{1/\left[ K : \Q \right]}$.  

\subsection{Strategy}
We will first lay down a few results that will provide us with a strategy to show that for any triple $n,a,b$ that satisfies different conditions, the Galois group of any factor of $f$ is symmetric.
\begin{lemma}
  \label{lem:inertia_group_acts_as_transposition}
  For any $p \not | ab$, the action of the inertia group $I\left( p \right)$ on $\Hom_{\Q}\left( \Q\left[ x \right] / \left( f \right), \overline{\Q} \right)$ has at most two non-fixed points. 
\end{lemma}
\editorial{It is a generalization of Martin's proof (\cite{MARTIN2004213})}
\begin{proof}
  If $I\left( p \right)$ acts trivially, we are done.
  Assume that it does not.
  It suffices to show that the roots of $f$ has a single orbit under $I\left( p \right)$, and that the length of this orbit is $2$.
  If two roots are in the same orbit under $I\left( p \right)$, then they have the same reduction modulo $p$.
  Therefore, if we denote by $\overline{f} \in \F_p\left[ x \right]$ the reduction of $f$ modulo $p$, then it suffices to show that $f$ has exactly one root in $\overline{\F_p}$ of multiplicity larger than $1$ and that its multiplicity is in fact $2$.

  Firstly, note that by \cite[Theorem 2]{Swan62} 
  \[
    \d\left( f \right) = 
    \left( -1 \right)^{\frac{n(n-1)}{2}} \left(  
      n^nb^{n-1} + \left( 1 - n \right)^{n - 1} a^n
    \right)
  \]
  and hence, since $p | \d\left( f \right)$ and since $p \nmid ab$, it follows that $p$ divides $n$ iff it divides $\left( n-1 \right)$ and thus it divides neither.
  
  Next, since 
  \begin{align*}
    & f' = n x^{n-1} + a \\
    & f'' = n \left( n - 1 \right) x^{n-2}
  \end{align*}
  and since $p \nmid n \left( n - 1 \right), a$ it follows that $f', f''$ have no shared root in $\overline{\F_p}$ and hence $f$ has no triple roots in $\overline{\F_p}$.

  Finally, note that any double root of $f$ is also a solution of the linear equation
  \[
    g := 
    n f - x f' = 
    nx^n + n a x + n b - n x^n - a x = 
    a \left( n - 1 \right) x + b n
  \]
  and hence since $p$ does not divide $a \left( n - 1 \right)$ there is exactly one double root of $f$ in $\overline{\F_p}$.
\end{proof}
\begin{corollary}
  \label{cor:inertia_group_acts_as_transposition}
  Let $g$ be an irreducible factor of $f$, then for any $p \not | ab$, the action of the inertia group $I\left( p \right)$ on $\Hom_{\Q}\left( \Q\left[ x \right] / \left( g \right), \overline{\Q} \right)$ has at most two non-fixed points.
\end{corollary}
\begin{proof}
  Let 
  \[
    \pi: \Q\left[ x \right] / \left( f \right) \to \Q\left[ x \right] / \left( g \right)
  \]
  be the quotient map.
  Then the precomposition map 
  \[
    \pi^\ast : \Hom_{\Q}\left( \Q\left[ x \right] / \left( g \right), \overline{\Q} \right) \to 
    \Hom_{\Q}\left( \Q\left[ x \right] / \left( f \right), \overline{\Q} \right)
  \]
  is a monomorphism of $G$-sets and hence the result follows from the previous lemma - \ref{lem:inertia_group_acts_as_transposition}.
\end{proof}
\begin{corollary}
  \label{cor:ramification_only_over_primes_dividing_a}
  Any number field $K$ that embeds non-surjectively into $\Q\left[ x \right] / \left( g \right)$ for any irreducible factor $g$ of $f$ is ramified only over primes dividing $a b$.
\end{corollary}
\begin{proof}
  It suffices to show that $K$ is unramified above any prime $p$ that does not divide $a b$.
  Let $p$ be such a prime.
  It suffices to show that $I\left( p \right)$ acts trivially on $\Hom_{\Q}\left( K, \overline{\Q} \right)$.
  Choose an embedding $\iota: K \hookrightarrow \Q\left[ x \right] / \left( g \right)$.
  Then the precompositio map 
  \[
    \iota^\ast : \Hom\left( \Q\left[ x \right] / \left( g \right), \overline{\Q} \right) \to 
    \Hom_{\Q}\left( K, \overline{\Q} \right)
  \]
  is a surjective $G$-equivariant map.
  Thus, $I\left( p \right)$ acts trivially on $\Hom_{\Q}\left( K, \overline{\Q} \right)$ since any non-trivial action of $I\left( p \right)$ on $\Hom_{\Q}\left( K, \overline{\Q} \right)$ would lift to an action on $\Hom_{\Q}\left( \Q\left[ x \right] / \left( g \right), \overline{\Q} \right)$ that would have at least $4$ non-fixed points, contradicting Corollary \ref{cor:inertia_group_acts_as_transposition}.
\end{proof}

\begin{lemma}
  A transitive permutation group that is generated by transpositions is the full symmetric group.
\end{lemma}
\begin{proof}
  \editorial{Sketch inspired by a discussion with Itamar Israeli:}

  There is a seuqnece of transpositions that connects any two elements.
  Given a product of a pair of transpositions 
  \[
    \left( a \; b \right) \left( b \; c \right) = \left( a \; b \; c \right)
  \]
  it can be shoretned to 
  \[
    \left( b \; c \right) \left( a \; b \right) \left( b \; c \right) = \left( a \; c \right) 
  \]
\end{proof}
\begin{corollary}
  \label{cor:primitive_group_with_transposition_is_full_symmetric}
  A primitive permutation group that contains a transposition is the full symmetric group.
\end{corollary}
\begin{proof}
  \editorial{Slightly inspired by Borys's thesis:}

  Let $H$ be a primitive permutation group that contains a transposition $\tau$.
  Let $K$ be the normal closure of $\left< \tau \right>$ inside $H$.
  Then the orbits of $K$ form a system of blocks for $H$.
  Therefore, $K$ is itself transitive and it is generated by transpositions and hence by the previouse lemma $K = S_n$ and hence $H = S_n$ as required.
\end{proof}
\begin{corollary}
  \label{cor:galois_group_with_transposition_ando_intermediate_extensions_is_full_symmetric}
  If the galois group $G_L$ of a field $L$ of degree $n$ contains a transposition and $L$ has no proper intermediate extensions, then $G_L = S_n$.
\end{corollary}
\begin{proof}
  According to Corollary \ref{cor:primitive_group_with_transposition_is_full_symmetric} it suffices to show that $G_L$ is primitive.
  Assume that it is not.
  Then there is a strictly smaller non-trivial $G$ set $S$ and a surjective $G$-equivariant map
  \[
    \pi : \Hom_{\Q}\left( \Q\left( \alpha \right), \overline{\Q} \right) \to S
  \]
  In particular, $\pi$ is an epimorphism of $G$-sets, and hence by the famous equivalnece of categories between $G$-sets and Etale algebras over $\Q$, $\pi$ corresponds to a monomorphism of Etale algebras over $\Q$, i.e. a field embedding 
  \[
    \iota : K \hookrightarrow L
  \]
  and since $S$ is non-trivial and strictly smaller, $K$ is a proper subextension of $L/\Q$, contradicting the assumption that $L$ has no proper intermediate extensions.
\end{proof}
\begin{corollary}
  \label{cor:the_almost_useful_criterion}
  If $g$ is an irreducible factor of $f$, there is a prime that divides the discriminant of $\Q\left[ x \right] / \left( g \right)$ but does not divide $a b$, and $\Q\left[ x \right] / \left( g \right)$ has no proper intermediate extensions, then the Galois group of $g$ is $S_{\deg g}$.
\end{corollary}
\begin{proof}
  Let $p$ be a prime that divides the discriminant of $\Q\left[ x \right] / \left( g \right)$ but does not divide $a b$.
  Then by Corollary \ref{cor:inertia_group_acts_as_transposition}, $I\left( p \right)$ acts on $\Hom_{\Q}\left( \Q\left[ x \right] / \left( g \right), \overline{\Q} \right)$ as a single transposition and hence the Galois group of $g$ contains a transposition.
  The result the follows from Corollary \ref{cor:galois_group_with_transposition_ando_intermediate_extensions_is_full_symmetric}.
\end{proof}
\begin{corollary}
  \label{cor:the_useful_criterion}
  Let $g$ be an irreducible factor of $f$, and assume that for every subfield $\Q \subsetneq K \subseteq \Q\left[ x \right] / \left( g \right)$ there is a prime that does not divide $a b$ over which $K$ ramifies, then the Galois group of $g$ is $S_{\deg g}$.
\end{corollary}
\begin{proof}
  According to Corollary \ref{cor:ramification_only_over_primes_dividing_a}, $\Q\left[ x \right] / \left( g \right)$ has no non-trivial subfields, and hence, since its discriminant is divisible by a prime that does not divide $a$, the Galois group of $g$ is $S_{\deg g}$ by Corollary \ref{cor:the_almost_useful_criterion}.
\end{proof}

Our strategy would be to find conditions on $n,a,b$ that would guarantee the conditions of the last corollary for any irreducible factor $g$ of $f$.

\subsection{$a b = \pm 1$}
Assume that $ab = \pm 1$.
\begin{corollary}
  The Galois group over $\Q$ of any irreducible factor of $f$ is symmetric.
\end{corollary}
\begin{remark}
  Selmer \cite{SelmerIrreducibility1956} noted that $f$ can always be reduced to either of the forms 
  \[
    x^n - x - 1 \quad \text{or} \quad x^n + x + 1 
  \]
  by substituting $x$ by $-x$.
  Moreover he showed that the first form is always irreducible, while the second form is irreducible whenever $n \not \equiv 2 \mod 3$ and factors into $x^2 + x + 1$ times an irreducible factor otherwise.
  When $x^n \pm x + 1$ is irreducible, the symmetry of its Galois group follows for example from \cite[Theorem 1]{Osada1987TheGG}.
  I did not find a reference for the reducible case.
\end{remark}
\begin{proof}
  This is immediate from Corollary \ref{cor:the_useful_criterion} since there is no prime that divides $a b \pm 1$ and the discriminant of any non-trivial extension of $\Q$ is divisible by some prime as a result of the Minkowski bound.
\end{proof}

\subsection{Strategy for slightly harder cases}

\begin{lemma}
  \label{lem:valuation_of_rd_of_subextensions}
  Let $p$ be a prime and $\Q \subseteq K \subseteq L$ number fields.
  Then 
  \[
    v_p\left( \rd K \right) \leq v_p \left( \rd L \right)
  \]
  and the inequality is strict iff $L/K$ is ramified over $p$.
\end{lemma}
\begin{proof}
  Indeed, 
  \[
    \disc\left( L \right) = \disc\left( K \right)^{\left[ L : K \right]} \Norm_{K / \Q}\left( \disc\left( L / K \right) \right)
  \]
  and hence 
  \begin{align*}
    & 
    v_p\left( \disc\left( L \right) \right) = \\
    & \left[ L : K \right] v_p\left( \disc\left( K \right) \right) + v_p\left( \Norm_{K / \Q}\left( \disc\left( L / K \right) \right) \right) \geq \\
    & \left[ L : K \right] v_p\left( \disc\left( K \right) \right)
  \end{align*}
  and hence
  \begin{align*}
    & v_p \left( \rd K \right) = \\
    & \frac{v_p\left( \disc\left( K \right) \right)}{\left[ K : \Q \right]} \leq \\
    & \frac{v_p\left( \disc\left( L \right) \right) / \left[ L / K \right]}{\left[ K : \Q \right]} = \\
    & \frac{v_p\left( \disc\left( L \right) \right)}{\left[ L : \Q \right]} = \\
    & v_p\left( \rd L \right)
  \end{align*}
  when the inequality in both of the above blocks is sharp iff $L/K$ is ramified over $p$ 
\end{proof}
It turns out that in some cases, all that we need is the combined strength of Corollary \ref{cor:the_useful_criterion} and Lemma \ref{lem:valuation_of_rd_of_subextensions} to show that the Galois group of $g$ is symmetric.
Namely, in those cases we can use the following corollary: 
\begin{corollary}
  If there are no number fields $K$ that are ramified only over primes dividing $a b$ and such that $v_p\left( \rd K \right) < v_p\left( \rd L \right)$ for all primes $p$ dividing $a b$, then the Galois group of $g$ is $S_{\deg g}$.
\end{corollary}


\subsection{$a = \pm 2, b = 1$}

We will handle the case $a = -2$ first.
Note that in this case $1$ is a root of $f$, and if we divide $f$ by $x-1$ we obtain the polynomial 
\begin{equation}
  \label{eq:the_polynomial_for_a_equal_minus_2}
  g := x^{n-1} + x^{n-2} + \ldots + x - 1
\end{equation}
After substituting $x$ by $1/x$ and multiplying by $-x^{n-1}$ we obtain the polynomial 
\[
  x^{n-1} - x^{n-2} - \ldots - x - 1
\]
which defines the same ring extension over $\Q$ as $g$ and which is none other than the characteristic polynomial of the generalized Fibonacci sequence which was studied by Wolfram \cite{Wolfram1998}.
Wolfram showed that this polynomial is irreducible over $\Q$, and Martin \cite{MARTIN2004213} showed that whenever $n$ is odd, its Galois group is the full symmetric group.
We will show that its Galois group is the full symmetric group for any $n$.
However, we will stick to our version of the polynomial defined in \eqref{eq:the_polynomial_for_a_equal_minus_2}.

We will denote by $d$ the discriminant of $g$. 
Then:

\begin{lemma}
  \label{lem:2-adic-valuation-of-discriminant}
\[
  v_2\left( d \right) = \begin{cases}
    n - 2 & n \text{ is even} \\
    0 & n \text{ is odd}
  \end{cases}
\]
\end{lemma}
\begin{proof}
  First of all note that 
  \begin{align*}
    & g'\left( \alpha \right) = \\
    & \prod_{\beta \in Z\left( g \right) - \left\{ \alpha \right\}}\left( \alpha - \beta \right) = \\
    & \frac{
      \prod_{\beta \in Z\left( f \right) - \left\{ \alpha \right\}}\left( \alpha - \beta \right)
    }{
      \alpha - 1
    } = \\
    & \frac{
      f'\left( \alpha \right)
    }{
      \alpha - 1
    } = \\
    & \frac{
      n \alpha^{n-1} - 2
    }{
      \alpha - 1
    } = \\
    & \frac{
      n \frac{2\alpha - 1}{\alpha} - 2
    }{
      \alpha - 1
    } = \\
    & \frac{
      2n - n\alpha^{-1} - 2
    }{
      \alpha - 1
    }
  \end{align*}
  Therefore, 
  \[
    v_2\left( d \right) = 
    v_2\left( N^L_\Q\left( 2n - n\alpha^{-1} - 2\right) \right) - 
    v_2\left( N^L_\Q\left( \alpha - 1 \right) \right) 
  \]
  We will start for the second summand.
  The norm of $\alpha - 1$ is the constant term of $g\left( x + 1 \right)$ which is $g\left( 1 \right) = n-2$ and hence $v_2\left( N^L_\Q\left( \alpha - 1 \right) \right) = v_2\left( n - 2 \right)$.
  If $n$ is odd, then $v_2\left( 2n - n\alpha'^{-1} - 2 \right) = 0$ for any conjugate $\alpha'$ of $\alpha$ in $\overline{\Q_2}$, and hence $v_2\left( N^L_\Q\left( 2n - n\alpha^{-1} - 2 \right) \right) = 0$ and hence 
  \[
    v_2\left( d \right) = 0 - v_2\left( n - 2 \right) = 0
  \]
  If $v_2\left( n \right) > 1$ then $v_2\left( 2n - n\alpha'^{-1} - 2 \right) = 1$ for any conjugate $\alpha'$ of $\alpha$ in $\overline{\Q_2}$, and hence $v_2\left( N^L_\Q\left( 2n - n\alpha^{-1} - 2 \right) \right) = n-1$ and hence
  \[
    v_2\left( d \right) = n - 1 - v_2\left( n - 2 \right) = n - 2
  \]
  Assume now that $v_2\left( n \right) = 1$.
  Note that in this case 
  \[
    f \equiv x^n + 1 \equiv \left( x^{n/2} + 1 \right)^2 \mod 2 
  \]
  and hence $f$ has exactly two roots lying above $1$ in $\overline{\Q_2}$, and since one of then is $\alpha$, then $g = \frac{f}{x-1}$ has exactly one root lying above $1$ in $\overline{\Q_2}$.
  Denote this root by $\alpha_0$.
  Then for any conjugate $\alpha'$ of $\alpha$ in $\overline{\Q_2}$,  
  \begin{align*}
    & v_2\left( 2n - n\alpha'^{-1} - 2 \right) = \\
    & v_2\left( 2n\alpha' - n - 2\alpha' \right) = \\
    & v_2\left( \left( \alpha' - 1 \right)n + \left( n - 2 \right)\alpha' \right) = \left( \ast \right)
  \end{align*}
  If $\alpha' \neq \alpha_0$, then $v_2\left( \alpha' - 1 \right) = 0$ and hence 
  \[
    v_2\left( n - 2 \right) \geq 2 > 1 = v_2\left( \left( \alpha' - 1 \right)n \right)
  \]
  and hence $\left( \ast \right) = 1$.
  If $\alpha' = \alpha_0$, then $v_2\left( \alpha' - 1 \right)$ is again the $2$-adic valuation of the constant term of $g\left( x + 1 \right)$, since $\alpha_0 - 1$ is the only root of $g\left( x + 1 \right)$ lying above $0$ in $\overline{\Q_2}$, and hence $v_2\left( \alpha' - 1 \right) = v_2\left( g\left( 1 \right) \right) = v_2\left( n - 2 \right)$.
  Therefore, in this case 
  \[
    v_2\left( \left( \alpha' - 1 \right)n \right) = v_2\left( n - 2 \right) + 1 > v_2\left( n - 2 \right) = v_2\left( \left( n - 2 \right)\alpha' \right)
  \]
   and hence $\left( \ast \right) = v_2\left( n - 2 \right)$.
   Thus,
  \begin{align*}
    & v_2\left( N^L_\Q\left( 2n - n\alpha^{-1} - 2 \right) \right) =
    \sum_{\alpha' \in Z\left( g \right)} v_2\left( 2n - n\alpha'^{-1} - 2 \right) = \\
    & \sum_{\alpha' \in Z\left( g \right) - \left\{ \alpha_0 \right\}} v_2\left( 2n - n\alpha'^{-1} - 2 \right) + v_2\left( 2n - n\alpha_0^{-1} - 2 \right) = \\
    & \sum_{\alpha' \in Z\left( g \right) - \left\{ \alpha_0 \right\}} 1 + v_2\left( n - 2 \right) = n - 2 + v_2\left( n - 2 \right)
  \end{align*}
  and hence \[
    v_2\left( d \right) = n - 2 + v_2\left( n - 2 \right) - v_2\left( n - 2 \right) = n - 2
  \]
  as claimed.
\end{proof} 
\begin{corollary}
  \label{cor:the_galois_group_of_x^n-2x+1}
  The Galois group of the non-linear factor of $x^n - 2x + 1$ is $S_{n-1}$.
\end{corollary}
\begin{proof}
  Let $\Q \subsetneq K \subseteq \Q\left[ x \right] / \left( g \right)$ be a sub-extension.
  Then according to Lemma \ref{lem:valuation_of_rd_of_subextensions} 
  \[
    v_2\left( \rd K \right) \leq v_2\left( \rd\left( \Q\left[ x \right] / \left( g \right) \right) \right) \leq \frac{v_2\left( g \right)}{n - 1}  \leq
    \frac{n - 2}{n - 1} < 1
  \]
  Therefore, if $\d\left( K \right)$ is divisible only by $2$, then $\rd\left( K \right) < 2$.
  However, there are no number fields with root discriminant smaller than $2$ that are ramified only over $2$ \editorial{we can prove it either with the Minkowski bound as I did in the first version of the proof, or just by citing \url{https://www.lmfdb.org/NumberField/?rd=-2-2} like we do in the rest of the cases}.
  Therefore, the discriminant of $K$ is divisible by a prime that does not divide $a$ and hence the Galois group of $g$ is $S_{n-1}$ by Corollary \ref{cor:the_useful_criterion}.
\end{proof}

\begin{lemma}
  Assume that $n$ is odd.
  Then $x^n + 2x + 1$ is irreducible over $\Q$.  
\end{lemma}
\begin{proof}
  First of all note that $x^n + 2x + 1$ has no roots on the unit circle in $\C$.
  Indeed, if $x$ is such a root, then 
  \[
    1 = \left| 2x + 1 \right| \geq \left| 2x \right| - \left| 1 \right| = 1 
  \]
  and hence $x = -1$.
  However, 
  \[
    \left( -1 \right)^n + 2 \left( -1 \right) + 1 = 
    -1 -2 + 1 = -2 \neq 0
  \]
  Consider now the polynomial 
  \[
    x^n \left( \left( 1/x \right)^n + 2 \left( 1/x \right) + 1 \right) =
    x^n + 2x^{n-1} + 1
  \]
  whose roots are exactly the inverses of the roots of $x^n + 2x + 1$.
  For any small enough $\epsilon > 0$, the winding number of $x^n + 2x^{n-1} + 1$ around $0$ as $x$ traverses the circle of radius $1 + \epsilon$ is $n - 1$.
  Indeed, for such $\epsilon$, 
  \[
    |2x^{n-1}| > |x^n| + 1 \geq |x^n + 1|
  \]
  since 
  \[
    \left( 1+\epsilon \right)^n - 2\left( 1+\epsilon \right)^{n-1} + 1 < 0
  \]
  since 
  \[
    \frac{d}{dx}\left( x^n - 2x^{n-1} + 1 \right) |_{x=1} = n - 2\left( n - 1 \right) = 2 - n < 0
  \]
  However, since $x^n + 2x + 1$ has no roots on the unit circle, and hence so does $x^n + 2x^{n-1} + 1$, it follows that all of these $n - 1$ roots are in the interior of the unit circle.
  In other words, only one of its roots has absolute value greater than or equal to $1$ (in fact greater).
  If we assume towards contradiction that $x^n + 2x^{n-1} + 1$ is reducible over $\Q$, then since it is monic, it is reducible over $\Z$ as well, and hence one of its factors is a monic polynomial with integer coefficients whose roots are all of absolute value strictly less than $1$ which is a contradiction (because then for example the norm, which is the free coefficient up to a sign, is less than $1$ in absolute value).
  Therefore, $x^n + 2x^{n-1} + 1$ is irreducible over $\Q$ and hence so is $x^n + 2x + 1$ as required.
\end{proof}

\begin{corollary}
  $x^n + 2x + 1$ is irreducible whenever $n$ is odd and factors into $x + 1$ times an irreducible factor otherwise.
  The Galois group of any irreducible factor $g$ of $f = x^n + 2x + 1$ is $S_{\deg g}$.
\end{corollary}
\begin{proof}
  Assume first that $n$ is even, then 
  \[
    \left( -x \right)^n + 2 \left( -x \right) + 1 = 
    x^n - 2x + 1
  \]
  and hence by the above analysis, $x^n -2x + 1$ factors into $x + 1$ times an irreducible factor whose Galois group is $S_{n-1}$.
  
  Assume now that $n$ is odd.
  According to \cite[Theorem 2]{Swan62} 
  \[
    \d\left( f_{n,2} \right) = 
    \left( -1 \right)^{\frac{n(n-1)}{2}} \left(  
      n^n + \left( 1 - n \right)^{n - 1} 2^n
    \right)
  \]
  and hence $v_2\left( \d\left( f_{n,2} \right) \right) = 0$.
  Therefore, discriminant of any intermediate extension $\Q \subsetneq K \subseteq \Q\left[ x \right] / \left( f_{n,2} \right)$ is, odd and in particular it is divisible by a prime that does not divide $2$ and hence the Galois group of $f_{n,2}$ is $S_n$ by Corollary \ref{cor:the_useful_criterion}.
\end{proof}

\begin{strategy}
  \label{strategy:harder_cases}
  Let $p$ be a prime that divides $a b$.
  Using \url{https://www.lmfdb.org/NumberField/} for example, we can enumerate the first few number fields that are ramified only above primes dividing $a b$ in ascending order of the $p$-valuation of their root discriminant.
  Thus, our strategy would be to compose a list of properties of number fields, such that any number field that satisfies one of these properties, cannot be embedded into $L$ for any value of $n$ (when $a$ and $b$ are fixed).
  Assume that $C_p = C_p\left( a,b \right)$ is the smallest number such that there is a number field $K$ that is ramified only above primes dividing $a b$, does not satisfy any of the properties in our list, and has $v_p\left( \rd K \right) = C_p$.
  Then the Galois group of $L$ may be non-symmetric only if $v_p\left( \rd L \right) \geq C_p$.
  Indeed, if it is non-symmetric, then by corollary \ref{cor:the_useful_criterion}, there is some subextension $\Q \subsetneq K \subseteq L$ that is ramified only over primes dividing $a b$, and hence by the definition of $C_p$, $v_p\left( \rd K \right) \geq C_p$ and hence by Lemma \ref{lem:valuation_of_rd_of_subextensions}, $v_p\left( \rd L \right) \geq v_p\left( \rd K \right) \geq C_p$ as claimed.
\end{strategy} 
To compose such a list of properties, we will first prove the following:
\begin{lemma}
  \label{lem:imprimitivity_finite_obstruction}
  Let $K$ be a number field.
  Assume that there is an integer $m \in \Z$ and a prime $p$ not dividing $a b$ such that $K$ has no embedding into $\Q_p$ and 
  \[
    v_p\left( f\left( m \right) \right) = 1
  \]
  Then $K$ cannot be a subfield of $\Q\left[ x \right] / \left( f \right)$.
\end{lemma}
\begin{proof}
  Since $p$ does not divide $a$, it does not ramify in $K$ according to Corollary \ref{cor:ramification_only_over_primes_dividing_a}, and hence since $K$ has no embedding into $\Q_p$, $K$ has no embedding into any purely ramified extension of $\Q_p$.
  Therefore, to show that $K$ can not be a subfield of $\Q\left[ x \right] / \left( f \right)$, it suffices to show that $f$ has a root that is contained in a purely ramified extension of $\Q_p$.
  Indeed, let $e$ be the multiplicity of $m$ as a root of $f$ modulo $p$.
  Then there is a factorization 
    \[
      \pi\left( f \right) = \left( x-m \right)^e \overline{g} \in \F_p\left[ x \right]
    \]
    for some $\overline{g} \in \F_p\left[ x \right]$ that is coprime to $\left( x-m \right)$ where $\pi : \Z_p\left[ x \right] \to \F_p\left[ x \right]$ is the quotient map.
    By hensel's lemma, there is a factorization
    \[
      f = h g \in \Z_p\left[ x \right]
    \]
    where $\pi\left( h \right) = \left( x-m \right)^e$ and $\pi\left( g \right) = \overline{g}$.
    Consider now the polynomial 
    \[
      f\left( x + m \right) = 
      h\left( x + m \right)g\left( x + m \right) = x^n + \sum_{i=0}^{n-1} c_i x^i \in \Z\left[ x \right]  
    \]
    Then 
    \[
      v_p\left( c_0 \right) = v_p\left( f\left( m \right) \right) = 1
    \]
    and 
    \[
      \pi\left( f\left( x + m \right) \right) = 
      \pi\left( h\left( x + m \right) \right) \pi\left( g\left( x + m \right) \right) =
      x^e \overline{g}\left( x + m \right) \in \F_p\left[ x \right]
    \]
    and hence 
    \[
      v_p\left( c_i \right) \geq 1
    \]
    for all $i < e$.
    Therefore, the Newton polygon of $f\left( x + m \right)$ with respect to $p$ has a single segment of negative and its slope is at most $1/e$ in absolute value.
    Therefore, $f\left( x + m \right)$ has a root, call it $\beta$, whose $p$-valuation is at most $1/e$.
    Therefore, since the set of roots of $f\left( x + m \right)$ that reduce to $0$ modulo $p$ is exactly the set of roots of $h\left( x + m \right)$, whose degree is $e$, it follows that the ramification degree of $\beta$ over $\Q_p$ is exactly its over all algebraic degree over $\Q_p$ and hence $\Q_p\left( \beta \right) / \Q_p$ is purely ramified.
    This shows that there is an embedding, in particular the one extending the assignment $x \mapsto \beta$, of $\Q\left[ x \right] / \left( f \right)$ into a purely ramified extension of $\Q_p$  
\end{proof}

\begin{corollary}
  \label{cor:conditions_that_guarantee_full_galois_group}
  Assume that that for every number field $K$ that is ramified only over primes dividing $a b$ and whose root discriminant is less than $C_p\left( a,b \right)$, either of the following holds:
  \begin{enumerate}
    \item $K$ is purely imaginary and $g$ has a real root.
    \item There is a prime $p$ not divide $a$ such that $v_p\left( b + a + 1 \right) = 1$ and such that there is a polynomial $h$ defining $K$ over $\Q$ that has no solution mod $p$.
    \item $K$ is of even degree and there is a prime $p$ such that $v_p\left( b - a + 1 \right) = 1$ and such that there is a polynomial defining $K$ over $\Q$ that has no solution mod $p$.
    \item $v_p\left( \rd K \right) = v_p\left( \rd L \right)$ and the ramification degree of $p$ in $K$ is strictly smaller than the ramification degree of $p$ in $L$.
  \end{enumerate}
  Then the Galois group of $g$ is $S_{\deg g}$ if $v_p\left( \rd L \right) < C_p\left( a,b \right)$ for every prime $p$ dividing $a b$.
\end{corollary}
\begin{proof}
  Based on Strategy \ref{strategy:harder_cases}, it suffices to show that any field $K$ that satisfies one of the above conditions cannot be a subfield of $\Q\left[ x \right] / \left( g \right)$. 
  \begin{enumerate}
    \item Clear.
    \item Assume that the second condition holds.
    Since $h$ has no solution mod $p$, $K$ has no embedding into $\Q_p$ and hence since 
    \[
      1 = v_p\left( b + a + 1 \right) = v_p\left( f\left( 1 \right) \right)
    \]
    it follows from Lemma \ref{lem:imprimitivity_finite_obstruction} that $K$ cannot be a subfield of $\Q\left[ x \right] / \left( f \right)$. 
    \item Assume that the third condition holds.
    If $n$ is odd, then $K$ cannot be a subfield of $\Q\left[ x \right] / \left( f \right)$ since the latter has an odd degree.
    Assume then that $n$ is even.
    Again, since $h$ has no solution mod $p$, $K$ has no embedding into $\Q_p$ and hence since 
    \[
      1 = v_p\left( b - a + 1 \right) = v_p\left( f\left( -1 \right) \right)
    \]
    it follows from Lemma \ref{lem:imprimitivity_finite_obstruction} that $K$ cannot be a subfield of $\Q\left[ x \right] / \left( f \right)$.
    \item Assume that the fourth condition holds.
    Then by Lemma \ref{lem:valuation_of_rd_of_subextensions}, $K$ cannot be a subfield of $\Q\left[ x \right] / \left( f \right)$ since if it were, then $v_p\left( \rd K \right) < v_p\left( \rd L \right)$, contradicting the assumption of the fourth condition.
  \end{enumerate}
\end{proof}

\subsection{$2 \leq \left| b \right|$ square free and $\left| a \right| = \left| b \right|^d$ for some $d \geq 1$}
Assume that $2 \leq \left| b \right|$, that $b$ is square free and that $\left| a \right| = \left| b \right|^d$ for some $d \geq 1$.
Note that in this case $f$ is irreducible over $\Q$ since it is an Eisenstein polynomial with respect to the prime $2$.
\begin{lemma}
  \label{lem:valuation_of_rd_when_a_equal_b_is_square_free}
  Let $p$ be a prime dividing $a$.
  Then $v_p\left( \rd\left( L \right) \right) \leq d$.
\end{lemma}
\begin{proof}
  According to \cite{Swan62} 
  \[
    \d\left( f \right) = n^n b^{n-1} + \left( 1 - n \right)^{n - 1} a^n
  \]
  Hence, to bound $v_p\left( \d\left( f \right) \right)$, we will first assume towards a contradiction that
  \[
    v_p\left( n^n b^{n-1} \right) = v_p\left( \left( 1 - n \right)^{n - 1} a^n \right)
  \]
  However, this equality is equivalent to
  \[
    n v_p\left( n \right) + \left( n - 1 \right)v_p\left( b \right) = \left( n - 1 \right) v_p\left( 1 - n \right) + n v_p\left( a \right)
  \]
  which holds if and only if
  \[
    n v_p\left( n \right) - \left( n - 1 \right)v_p\left( n-1 \right) = n v_p\left( a \right) - \left( n - 1 \right)v_p\left( b \right) = n d - \left( n - 1 \right) = \left( d - 1 \right) n + 1 
  \]
  Since the left hand side is divisible by $p$, and if $d = 1$ then the right hand side is $1$, it follows that $d > 1$.
  Since the right hand side is positive, it follows that $n v_p\left( n \right) \geq \left( n - 1 \right)v_p\left( n - 1 \right) \geq 0$ and hence $n \equiv 0 \mod p$.
  However, it hence follows that the right hand side is congruent to $1$ modulo $p$, contradicting the fact that it is divisible by $p$.
  
  Therefore, 
  \begin{align*}
    & v_p\left( \d\left( L \right) \right) \leq \\
    & v_p\left( \d\left( f \right) \right) = \\
    & v_p\left( n^n b^{n-1} + \left( 1 - n \right)^{n - 1} a^n \right) = \\
    & \min\left\{ v_p\left( n^n b^{n-1} \right), v_p\left( \left( 1 - n \right)^{n - 1} a^n \right) \right\} = \\
    & \min\left\{ 
      n v_p\left( n \right) + \left( n - 1 \right), 
      \left( n - 1 \right) v_p\left( 1 - n \right) + dn 
    \right\} \leq d n
  \end{align*}
  and hence 
  \[
    v_p\left( \rd L \right) \leq d
  \]
  as claimed.
\end{proof} 

\begin{corollary}
  \label{cor:the_case_of_a_equal_b_equal_p}
  If there is no number field that is ramified only above $p$ and whose root discriminant is at most $p^d$ that does not satisfy any of the conditions in Corollary \ref{cor:conditions_that_guarantee_full_galois_group}, then $\Gal\left( f \right) = S_n$ for any $n$.
\end{corollary}
\begin{proof}
  Follows immediately from Corollary \ref{cor:conditions_that_guarantee_full_galois_group} and Lemma \ref{lem:valuation_of_rd_when_a_equal_b_is_square_free}.
\end{proof}

  \]
  is $S_n$ for any $n$.

\end{proof}

\subsection{$a = \pm 3, \pm 4, \pm 5, b = 1$}
\begin{lemma}
  If $\left| a \right| > 2$ then $f_{n,a}$ is irreducible over $\Q$.
\end{lemma}
\begin{proof}
  It suffices to show that 
  \[
    \widehat{f_{n,a}} := x^n f_{n,a}\left( 1/x \right) = x^n + ax^{n-1} + 1 
  \]
  is irreducible over $\Q$.
  For any $x \in S^1 \subseteq \C$, 
  \[
    \left| a x^{n-1} \right| = \left| a \right| > 2 = \left|x^n\right| + \left|1\right| \geq \left|x^n + 1\right|  
  \]
  and hence the winding number of $\widehat{f_{n,a}}\left( x \right)$ around $0$ as $x$ traverses the unit circle is equal to that of $- a x^{n-1}$ which is $n-1$.
  Therefore, $\widehat{f_{n,a}}$ has $n-1$ roots in the interior of the unit disk and hence it is irreducible over $\Q$.
\end{proof}
\begin{remark}
  The above lemma also follows from \cite[Theorem 1]{Ljungren1960}.
\end{remark}

\begin{lemma}
  \label{lem:valuation_of_disc}
  Let $p$ be a prime dividing $a$.
  If 
  \[
    v_p\left( n \right) \neq v_p\left( a \right) \text{ or } \left( n / a \right)^n \not \equiv -1 \mod p
  \]
  then 
  \[
    v_p\left( \disc\left( f_{n,a} \right) \right) = n \min\left\{  
      v_p\left( n \right), v_p\left( a \right) 
    \right\}
  \] 
\end{lemma}
\begin{proof}
  Assume first that $v_p\left( n \right) \neq v_p\left( a \right)$.
  According to \cite[Theorem 2]{Swan62}
  \[
    \disc\left( f_{n,a} \right) = 
    \left( -1 \right)^{\frac{n(n-1)}{2}} \left(  
      n^n + \left( 1 - n \right)^{n - 1} a^n
    \right)
  \]
  Then either $v_p\left( n \right) = 0$ and hence $v_p\left( \disc\left( f_{n,a} \right) \right) = 0$ or $v_p\left( n \right) > 0$ and hence $v_p\left( n - 1 \right) = 0$ and thus $v_p\left( \left( 1 - n \right)^{n - 1} a^n \right) = n v_p\left( a \right)$ and hence 
  \[
    v_p\left( \disc\left( f_{n,a} \right) \right) = 
    \min\left\{ v_p\left( n^n \right), v_p\left( \left( 1 - n \right)^{n - 1} a^n \right) \right\} = 
    n \min\left\{ v_p\left( n \right), v_p\left( a \right) \right\}
  \]
  
  Assume now that $v_p\left( n \right) = v_p\left( a \right)$.
  Then 
  \[
    \disc\left( f_{n,a} \right) = 
    \Res\left( f_{n,a}, f_{n,a}' \right) = 
    p^{n v_p\left( a \right)} \Res\left(  
      f_{n,a},
      p^{-v_p\left( a \right)} f_{n,a}'
    \right)
  \]
  Since $p^{-v_p\left( a \right)} f_{n,a}' \in \Z\left[ x \right]$ it follows that $v_p\left( \disc\left( f_{n,a} \right) \right) \geq n v_p\left( a \right)$ and equality holds iff $f_{n,a}$ and $p^{-v_p\left( a \right)} f_{n,a}'$ have no common root in $\overline{\F_p}$.
  Assume then that there is no equality, i.e. that 
  \begin{align*}
    & \overline{f_{n,a}} = x^n + ax + 1 = x^n + 1 \in \F_p\left[ x \right] \\
    & p^{-v_p\left( a \right)} \overline{f_{n,a}}' = n p^{-v_p\left( a \right)} x^{n-1} + a p^{-v_p\left( a \right)} \in \F_p\left[ x \right]
  \end{align*}
  have a common root.
  This root in particulat annahilates 
  \[
    g_{n,a} := 
    x \overline{f_{n,a}}' - n \overline{f_{n,a}} = 
    n x^n + a x - nx^n - n = 
    a x - n
  \]
  i.e. it is equal to $n / a$ and hence
  \[
    0 = \overline{f_{n,a}}\left( n / a \right) = \left( n / a \right)^n + 1
  \]
  and thus $\left( n / a \right)^n \equiv -1 \mod p$.
\end{proof}

Assume now that $a$ is a power of some prime $p$.
\begin{corollary}
  \label{cor:for_p_power_a} 
  If any number field that is ramified only over $p$ and whose root discriminant is at most $a$ in absolute value satisfies one of the conditions of corollary \ref{cor:conditions_that_guarantee_full_galois_group} then for any $n \in S\left( a \right)$, $\Gal\left( f \right) = S_n$.
\end{corollary}

\begin{corollary}
  For any $n \in S\left( 3 \right)$, $\Gal\left( f_{n,\pm3} \right) = S_n$.  
\end{corollary}
\begin{proof}
  According to \url{https://www.lmfdb.org/NumberField/?ram_quantifier=subset&ram_primes=3&rd=-3-3}, the only non-trvial number field that is ramified only over $3$ and whose root discriminant is at most $3$ is the cyclotomic extension $\Q\left( \zeta_3 \right) = \Q\left( \sqrt{-3} \right)$ which is purely imaginary and hence the result follows from Corollary \ref{cor:for_p_power_a}.
\end{proof}

\begin{corollary}
  For any $n \in S\left( 4 \right)$, $\Gal\left( f_{n,\pm4} \right) = S_n$.
\end{corollary}
\begin{proof}
  According to \url{https://www.lmfdb.org/NumberField/?ram_quantifier=subset&ram_primes=2&rd=-4-4&sort_order=rd&sort_order=rd}, the only non-trivial number fields that are ramified only over $2$ and whose root discriminant is at most $4$ are the three purely imaginary fields defined by the polynomials $x^2 + 1, x^2 + 2, x^4 + 1$, and the quadratic field $\Q\left( \sqrt{2} \right)$ which is defined by $x^2 -2$ which has no solution mod $3$ which divides $4 + 2 = -\left( -4 \right) + 2$, is inert, and hence the result follows from Corollary \ref{cor:for_p_power_a}.
\end{proof}

\begin{corollary}
  For any $n \in S\left( 5 \right)$, $\Gal\left( f_{n,\pm5} \right) = S_n$.
\end{corollary}
\begin{proof}
  According to \url{https://www.lmfdb.org/NumberField/?ram_quantifier=subset&ram_primes=5&rd=-5-5&sort_order=rd&sort_order=rd}, the only non-trivial number fields that are ramified only over $5$ and whose root discriminant is at most $5$ are the cyclotomic extension $\Q\left( \zeta_5 \right)$ and the quadratic field $\Q\left( \sqrt{5} \right)$.
  The first is purely imaginary, and the second may be defined by $x^2 - 5$ which has no solution either mod $7$ which divides $5 + 2$ or mod $3$ which divides $-5 + 2$ and hence the result follows from Corollary \ref{cor:for_p_power_a}.
\end{proof}
